**x，y，a，b；x->y**

```c++
#include <iostream>
using namespace std;

int main()
{
    int x, y, a, b;
    cin >> x >> y >> a >> b;
    if (x > y) swap(x, y);
    
    if (b % a > 0) cout << -1 << endl;
    else {
        int k = b / a;
        int n = 0;
        int l = 0, r = k;
        while (l < r) {
            int mid = (l + 1 + r) / 2;
            if (pow(y, mid) <= k) l=mid;
            else r=mid-1;
        }
        if (pow(y, l) <= k) n = l;
        else n = 0;
        // y最多能用n次
        long long ans = -1;
        for (int i = n; i >= 0; i--) {
            int delt = 0;
            if (i == 0) delt = k;
            else {
                int tmp = pow(y, i);
                if (k % tmp > 0) continue;
                else delt = k / tmp;
            }
            if (delt % x == 0) {
                ans = i + delt / x;
                break;
            }
        }
        cout << ans << endl;
    }

    return 0;
}
```
