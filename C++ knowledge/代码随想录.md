#### 二叉树

- 前序

[code](https://leetcode.cn/problems/binary-tree-preorder-traversal/)

```c++
vector<int> preorderTraversal(TreeNode* root) {
	vector<int> ans;
	stack<TreeNode*> stk;
	TreeNode *p = root;
	while(p || !stk.empty()) {
		while(p) {
			ans.push_back(p->val);
			stk.push(p);
			p = p->left;
		}
		p = stk.top();
		stk.pop();
		p = p->right;
	}
	return ans;
}
```

```c++
vector<int> inorderTraversal(TreeNode *root) {
	vector<int> ans;
	stack<TreeNode *> st;
	TreeNode *p = root;
	while (p || !st.empty()) {
		while (p) {
			st.push(p);
			p = p->left;
		}
		p = st.top();
		st.pop();
		ans.push_back(p->val);
		p = p->right;
	}
	return ans;
}
```

- 后序（通过pre指针维护后序遍历中的访问的上一个节点）

[code](https://leetcode.cn/problems/binary-tree-postorder-traversal/submissions/)

```c++
vector<int> postorderTraversal(TreeNode* root) {
    vector<int> ans;
    stack<TreeNode*> st;
    TreeNode* p = root;
    TreeNode* pre = nullptr;
    while(p || !st.empty()) {
        while(p) {
            st.push(p);
            p = p->left;
        }
        // 此时p == nullptr
        p = st.top();
        st.pop();
        if (p->right == nullptr || p->right == pre) {
            ans.push_back(p->val);
            pre = p;
            p = nullptr;
        }
        else {
            st.push(p);
            p = p->right;
        }
    }
    return ans;
}
```

# 数论

#### gcd（最大公约数）

```c++
//1、更相相减法
int gcd1(int x, int y) {
	while(1) {
		if(x>y) {
			x -= y;
		}
		else if (x<y) {
			y -= x;
		}
		else {
			return x;
		}
	}
}

//2、辗转相除法
int gcd2(int x, int y) {
	while(x*y) {
		if (x>y) {
			x %= y;
		}
		else if (y>x) {
			y %= x;
		}
	}
	return x>y ? x : y;
}
//3、结合两种方法使速度更快（递归）
int gcd(int data1,int data2)
{
    if (data1 == data2)
    {
        return data1;
    }
    //为了保证较大的数始终在前面，减少了代码
    if (data1 < data2)
    {
        gcd(data2,data1);
    }
    else
    {
        //与1操作是为了判断奇偶
        if (!(data1 & 1) && !(data2 & 1))//两数都是偶数
        {
            return gcd(data1>>1,data2>>1)<<1;       
        }
        else if(!(data1 & 1) && (data2 & 1))//data1为偶数，data2为奇数
        {
            return gcd(data1>>1,data2);
        }
        else if((data1 & 1) && !(data2 & 1))//data1为奇数，data2为偶数
        {
            return gcd(data1,data2>>1);
        }
        else//当两个数都为奇数时，应用更相减损法
        {
            return gcd(data2,data1 - data2);
        }
    }
}
```

#### 随机数等概率生成

> **大范围概率函数生成小范围概率函数**
>
> 如：`rand7()`随机生成0-6；如何等概率生成`rand5()`
>
> 直接使用一次`rand7()`，如果随机值是5,6，则直接丢弃；
>
> 分析：第一次选中x，第二次选中x，......，第n次选中x（x在0-4之间）
> $$
> \frac{1}{7} + \frac{1}{7}*(\frac{2}{7})^2 + \frac{1}{7}*(\frac{2}{7})^3 ...
> $$
> 求解可知0-4中任何一个数的概率都是 1/5
>
> **小范围概率函数生成大范围概率函数**
>
> 如：`rand7()` 随机等概率生成 `rand10()`
>
> - 先放大范围：`rand7()` 0~6 乘以 7；得到 0,7,14,21,28,35,42（等概率 1/7）
> - 在填充两数之间的范围：加上`rand7()`；0,1,2,3,4,5,6......（范围[0,48]，每个数字的概率为1/49）
> - 相当于`rand49()` 生成 `rand10()`
